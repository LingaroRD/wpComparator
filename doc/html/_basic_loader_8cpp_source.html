47 bool BasicLoader::locROI(const cv::Mat& mask,cv::Point& P1,cv::Point& P2)
48 {
49  int rows = mask.rows;
50  int cols = mask.cols;
51  cv::Mat_<cv::Vec3f> ImPoint = mask;
52  cv::Vec3f maskColor(0,128,0);
53  cv::Point pMax(0,0);
54  cv::Point pMin(cols,rows);
55  bool hasColor = false;
56  for (int i=0;i<rows;++i)
57  for (int j=0;j<cols;++j)
58  if (ImPoint(i,j)==maskColor)
59  {
60  hasColor = true;
61  if (pMax.x<j)
62  pMax.x=j;
63  if (pMax.y<i)
64  pMax.y=i;
65  if (pMin.x>j)
66  pMin.x=j;
67  if (pMin.y>i)
68  pMin.y=i;
69  }
70  P1 = pMin;
71  P2 = pMax;
72  return hasColor;
73 }

125 void BasicLoader::comparePages(const BasicLoaderAppData& pg) {
126  LayerDataVector originalWebPage,images;
127  cv::Point fP11,fP12,fP21,fP22;
128  cv::Mat tmp1,tmp2,*scannedWeb,*originalWeb;
129  /* layer 0 == not modified page */
130  acquireLayers(0,pg,originalWebPage);
131  /* get next layers */
132  for (int i=1;i<=m_layerNumber;++i) {
133  std::cout << "Layer number: " << i << std::endl;
134  /* Make screenshots for current script configuration */
135  acquireLayers(i,pg,images);
136  for (unsigned n=0;n<images.size();++n) {
137  /* Retrieve an unmodified image of web page for current mask */
138  if (!(originalWeb = getImage(images[n],originalWebPage)))
139  continue;
140  if (!locROI(images[n].image,fP11,fP12))
141  break;
142  tmp1 = (*originalWeb)(cv::Rect(fP11.x,fP11.y,fP12.x-fP11.x,fP12.y-fP11.y));
143  for (unsigned m=n+1;m<images.size();++m) {
144  if (!(scannedWeb = getImage(images[m],originalWebPage)))
145  continue;
146  if (!locROI(images[m].image,fP21,fP22))
147  continue;
148  tmp2 = (*scannedWeb)(cv::Rect(fP21.x,fP21.y,fP22.x-fP21.x,fP22.y-fP21.y));
149  ResultType rt;
150  /* Compare selected fragments */
151  rt.cmpCoeff=comparator(tmp1,tmp2);
152  rt.def1.image=tmp1.clone();
153  rt.def1.browser=images[n].browser;
154  rt.def2.image=tmp2.clone();
155  rt.def2.browser=images[m].browser;
156  rt.layerNumber = i;
157  m_results.push_back(rt);
158  }
159  }
160  }
161 }

211 void BasicLoader::run(AppData& appPaths) {
212  /* May throw bad_cast */
213  BasicLoaderAppData& appData = dynamic_cast<BasicLoaderAppData&>(appPaths);
214  /* Load default settings */
215  m_layerNumber = boost::lexical_cast<int>(m_options.getProperty("LayerNumber"));
216  m_counterInitial = boost::lexical_cast<int>(m_options.getProperty("CounterInitial"));
217  m_layerChangeDelay = boost::lexical_cast<int>(m_options.getProperty("LayerChangeDelay"));
218  /* Load script file */
219  const int MAX_SIZE = 300000;
220  char dataBuffer[MAX_SIZE]={0};
221  std::ifstream JS(m_jsScriptName);
222  if (!JS.is_open())
223  throw std::runtime_error(std::string("Unable to open: ")+m_jsScriptName+INSERT_POS_MARK());
224  JS.read(dataBuffer,sizeof(dataBuffer));
225  JS.close();
226  std::string currentJS(dataBuffer);
227  /* Replace all loaded parameters */
228  if (!replaceString(currentJS,"{LayerNumber}",m_options.getProperty("LayerNumber")))
229  throw std::runtime_error(std::string("Unable to substitute parameter: LayerNumber ")+INSERT_POS_MARK());
230  if (!replaceString(currentJS,"{CounterInitial}",m_options.getProperty("CounterInitial")))
231  throw std::runtime_error(std::string("Unable to substitute parameter: CounterInitial ")+INSERT_POS_MARK());
232  if (!replaceString(currentJS,"{LayerChangeDelay}",m_options.getProperty("LayerChangeDelay")))
233  throw std::runtime_error(std::string("Unable to substitute parameter: LayerChangeDelay ")+INSERT_POS_MARK());
234  /* Load tested HTML */
235  std::string fullPath = appData.working_directory+appData.file_name;
236  std::ifstream HTML(fullPath);
237  if (!HTML.is_open())
238  throw std::runtime_error(std::string("Unable to load: ")+appData.file_name+INSERT_POS_MARK());
239  HTML.read(dataBuffer,sizeof(dataBuffer));
240  HTML.close();
241  std::string currentHTML(dataBuffer);
242  /* Save mixed html and js*/
243  currentHTML+=currentJS;
244  std::ofstream outputFile(m_TempName,std::ios_base::out);
245  if (!outputFile.is_open())
246  throw std::runtime_error(std::string("Unable to write: ")+m_TempName+INSERT_POS_MARK());
247  outputFile << currentHTML;
248  outputFile.close();
249  /* Start selected browsers */
250  BrowserDataVector::iterator it = appData.browser_data.begin();
251  fullPath = appData.working_directory+m_TempName;
252  for (;it!=appData.browser_data.end();++it)
253  if (!start((*it),fullPath))
254  throw std::runtime_error(std::string("Unable to start: ")+(*it).path+INSERT_POS_MARK());
255  /* Start comparison */
256  comparePages(appData);
257  /* Close browsers */
258  it = appData.browser_data.begin();
259  for (;it!=appData.browser_data.end();++it)
260  closeBrowsers(*it);
261 }
